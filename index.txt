    function escapeHTML(str) {
        if (!str) return '';
        return str.replace(/[&<>'"]/g, tag => ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', "'": '&#39;', '"': '&quot;' }[tag]));
    }

    window.AppModal = {
        overlay: document.getElementById('customModal'),
        title: document.getElementById('modalTitle'),
        msg: document.getElementById('modalMessage'),
        input: document.getElementById('modalInput'),
        cancelBtn: document.getElementById('modalCancelBtn'),
        confirmBtn: document.getElementById('modalConfirmBtn'),
        callback: null,
        show(options, callback) {
            this.callback = callback;
            this.title.innerText = options.title || '알림';
            this.msg.innerText = options.msg || '';
            this.input.value = '';
            this.confirmBtn.innerText = options.confirmText || '확인';

            this.overlay.style.display = 'flex';
            if(options.type === 'prompt') setTimeout(() => this.input.focus(), 100);
        },

        submit() { this.close(); if (this.callback) this.callback(this.input.style.display === 'block' ? this.input.value : true); },

    window.toggleAdminPanel = function() {
        if (isAdmin) {

            document.getElementById('adminBtn').innerText = "⚙️ 선생님 로그인";
            document.getElementById('adminBtn').classList.remove('active');
            refreshAdminUI();

            if(box.style.display === 'block') document.getElementById('adminPwInput').focus();
        }
    };

        }

    function refreshAdminUI() {
        document.querySelectorAll('.reply-input-group').forEach(g => g.style.display = isAdmin ? 'flex' : 'none');
        document.querySelectorAll('.reply-actions').forEach(g => g.style.display = isAdmin ? 'flex' : 'none');
    }

    window.addQuestion = function() {
        const title = document.getElementById('titleInput').value.trim();
        const text = document.getElementById('questionInput').value.trim();
        const author = document.getElementById('authorInput').value.trim() || '익명'; 
        const password = document.getElementById('pwInput').value.trim();
        const isSecret = document.getElementById('secretInput').checked;
        
        if (!title || !password || !text) {
            window.AppModal.show({type: 'alert', msg: '제목, 비밀번호, 내용을 모두 입력해주세요.'});
            return;
        }
        push(questionsRef, { title, author, password, text, isSecret, time: getCurrentTime(), timestamp: Date.now() });
        document.getElementById('titleInput').value = '';
        document.getElementById('questionInput').value = '';
        document.getElementById('authorInput').value = '';
        document.getElementById('pwInput').value = '';
        document.getElementById('secretInput').checked = false;
    };

    window.togglePost = (id) => {
        const li = document.getElementById(`post-${id}`);
        const content = document.getElementById(`content-${id}`);
        const isSecret = li.dataset.secret === 'true';
        const correctPw = li.dataset.pw;

        if (content.style.display === 'block') {
            content.style.display = 'none';
            return;
        }

        if (isSecret && !isAdmin) {
            window.AppModal.show({type: 'prompt', title: '🔒 비밀글', msg: '작성 시 설정한 비밀번호를 입력해주세요.'}, (pw) => {
                if (pw === correctPw) {
                    content.style.display = 'block';
                } else if (pw !== null) {
                    window.AppModal.show({type: 'alert', msg: '비밀번호가 일치하지 않습니다.'});
                }
            });
        } else {
            content.style.display = 'block';
        }
    };

    window.handleEdit = function(id) {
        const li = document.getElementById(`post-${id}`);
        const correctPw = li.dataset.pw;
        if (!isAdmin) {
            window.AppModal.show({type: 'prompt', title: '수정', msg: '비밀번호를 입력해주세요.'}, (pw) => {
                if (pw === correctPw) enableEditMode(id);
                else if (pw !== null) window.AppModal.show({type: 'alert', msg: '비밀번호 불일치'});
            });
        } else enableEditMode(id);
    };

    function enableEditMode(id) {
        const bodyDiv = document.getElementById(`post-body-${id}`);
        const currentText = bodyDiv.dataset.text;
        const currentTitle = bodyDiv.dataset.title;
        bodyDiv.innerHTML = `
            <input type="text" id="edit-title-${id}" value="${escapeHTML(currentTitle)}" style="width:100%; padding:12px; margin-bottom:10px; border:1px solid #1877f2; border-radius:6px; font-weight:bold; outline:none; box-sizing:border-box;">
            <textarea id="edit-textarea-${id}" style="width:100%; height:80px; padding:12px; border:1px solid #1877f2; border-radius:6px; outline:none; resize:none; box-sizing:border-box;">${escapeHTML(currentText)}</textarea>
            <div style="display:flex; justify-content:flex-end; gap:8px; margin-top:10px;">
                <button onclick="window.cancelEdit('${id}')" style="padding:8px 16px; border:none; background:#e4e6eb; border-radius:6px; cursor:pointer;">취소</button>
                <button onclick="window.saveEdit('${id}')" style="padding:8px 16px; border:none; background:#1a73e8; color:white; border-radius:6px; cursor:pointer;">저장</button>
            </div>`;
    }

    window.cancelEdit = (id) => {
        const bodyDiv = document.getElementById(`post-body-${id}`);
        bodyDiv.textContent = bodyDiv.dataset.text;
    };

    // ★ 메인 게시글 저장 오류 해결: 수동으로 화면 업데이트 ★
    window.saveEdit = (id) => {
        const newTitle = document.getElementById(`edit-title-${id}`).value.trim();
        const newText = document.getElementById(`edit-textarea-${id}`).value.trim();
        if (newTitle && newText) {
            update(ref(db, `questions/${id}`), { title: newTitle, text: newText });
            
            // 데이터베이스 업데이트 후, UI의 텍스트 에리어를 없애고 즉시 새 텍스트로 반영
            const bodyDiv = document.getElementById(`post-body-${id}`);
            bodyDiv.dataset.text = newText;
            bodyDiv.dataset.title = newTitle;
            bodyDiv.textContent = newText;
            
            const li = document.getElementById(`post-${id}`);
            const isSecret = li.dataset.secret === 'true';
            document.getElementById(`title-text-${id}`).textContent = (isSecret ? '🔒 ' : '') + newTitle;
            
        } else {
            window.AppModal.show({type: 'alert', msg: '제목과 내용을 모두 입력해주세요.'});
        }
    };

        if (isAdmin) {

        } else {
            window.AppModal.show({type: 'prompt', msg: '비밀번호를 입력하세요.'}, (pw) => {
                if (pw === correctPw) window.AppModal.show({type: 'confirm', msg: '삭제하시겠습니까?', confirmText: '삭제'}, (res) => { if(res) remove(ref(db, `questions/${id}`)); });
                else if (pw !== null) window.AppModal.show({type: 'alert', msg: '비밀번호 불일치'});
            });
        }
    };

    // Firebase 연동
    onChildAdded(questionsRef, (snapshot) => {
        const id = snapshot.key;
        renderPost(id, snapshot.val());
        
        const repliesRef = ref(db, `questions/${id}/replies`);
        onChildAdded(repliesRef, (rep) => { 
            renderComment(id, rep.key, rep.val()); 
            updateReplyCount(id, 1); 
        });
        onChildRemoved(repliesRef, (rep) => {
            document.getElementById(`reply-${id}-${rep.key}`)?.remove();
            updateReplyCount(id, -1);
        });
        onChildChanged(repliesRef, (rep) => {
            const textContainer = document.getElementById(`reply-text-${id}-${rep.key}`);
            if (textContainer && !textContainer.querySelector('textarea')) {
                textContainer.textContent = rep.val().text;
                document.getElementById(`reply-${id}-${rep.key}`).dataset.text = rep.val().text;
            }
        });
    });

    onChildRemoved(questionsRef, (snapshot) => document.getElementById(`post-${snapshot.key}`)?.remove());

    onChildChanged(questionsRef, (snapshot) => {
        const id = snapshot.key;
        const data = snapshot.val();
        const titleSpan = document.getElementById(`title-text-${id}`);
        const bodyDiv = document.getElementById(`post-body-${id}`);
        
        const displayTitle = (data.isSecret ? '🔒 ' : '') + (data.title || data.text.substring(0, 20) + '...');
        if (titleSpan) titleSpan.textContent = displayTitle;
        
        if (bodyDiv && !bodyDiv.querySelector('textarea')) {
            bodyDiv.textContent = data.text;
            bodyDiv.dataset.text = data.text;
            bodyDiv.dataset.title = data.title;
        }
    });

    function renderPost(id, data) {
        const list = document.getElementById('qnaList');
        const li = document.createElement('li');
        li.className = 'board-item';
        li.id = `post-${id}`;
        li.dataset.pw = data.password || ''; 
        li.dataset.secret = data.isSecret || false; 
        
        const author = data.author || '익명';
        const rawTitle = data.title || data.text.substring(0, 20) + '...';
        const displayTitle = (data.isSecret ? '🔒 ' : '') + rawTitle;
        const listTime = data.time.split(' ')[0].substring(5) + ' ' + data.time.split(' ')[1];

        li.innerHTML = `
            <div class="item-row" onclick="window.togglePost('${id}')">
                <div class="item-title-wrapper">
                    <span class="item-title" id="title-text-${id}">${escapeHTML(displayTitle)}</span>
                    <span class="reply-count" id="reply-count-${id}" style="display:none;">[0]</span>
                </div>
                <div class="item-meta-mobile">
                    <div class="item-author">${escapeHTML(author)}</div>
                    <div class="item-time">${listTime}</div>
                </div>
            </div>
            <div class="item-content" id="content-${id}">
                <div class="post-actions-inline">
                    <button class="action-btn" onclick="window.handleEdit('${id}')">수정</button>
                    <button class="action-btn delete" onclick="window.handleDelete('${id}')">삭제</button>
                </div>
                <div id="post-body-${id}" class="post-body" data-text="${escapeHTML(data.text)}" data-title="${escapeHTML(rawTitle)}">${escapeHTML(data.text)}
                <div style="font-size:11px; color:#999; margin-top:10px;">작성일시: ${data.time}</div>
                </div>
                <div class="comment-section">
                    <div class="comment-header">💬 댓글</div>
                    <ul class="comment-list" id="comment-list-${id}"></ul>
                    <div class="reply-input-group" id="reply-group-${id}">
                        <textarea id="reply-input-${id}" placeholder="선생님 답변 입력..." onkeypress="if(event.key === 'Enter' && !event.shiftKey) { event.preventDefault(); window.addReply('${id}'); }"></textarea>
                        <button onclick="window.addReply('${id}')">등록</button>
                    </div>
                </div>
            </div>`;
        list.prepend(li);
        refreshAdminUI();
    }

    function updateReplyCount(qId, change) {
        const span = document.getElementById(`reply-count-${qId}`);
        if (!span) return;
        let current = parseInt(span.dataset.count || '0') + change;
        span.dataset.count = current;
        span.style.display = current > 0 ? 'inline' : 'none';
        span.textContent = `[${current}]`;
    }

    window.addReply = (id) => {
        if (!isAdmin) return;
        const input = document.getElementById(`reply-input-${id}`);
        const text = input.value.trim();
        if (text) push(ref(db, `questions/${id}/replies`), { text, time: getCurrentTime(), isTeacher: true });
        input.value = '';
    };

    function renderComment(qId, replyId, data) {
        const list = document.getElementById(`comment-list-${qId}`);
        const li = document.createElement('li');
        li.className = 'comment-item';
        li.id = `reply-${qId}-${replyId}`;
        li.dataset.text = data.text; 
        
        li.innerHTML = `
            <div class="comment-profile">👨‍🏫</div>
            <div class="comment-content">
                <div class="comment-header-row">
                    <div class="comment-meta">
                        <span class="teacher-badge">선생님</span>
                        <span class="comment-time">${data.time}</span>
                    </div>
                    <div class="reply-actions" style="display: ${isAdmin ? 'flex' : 'none'};">
                        <button class="action-btn" onclick="window.editReply('${qId}', '${replyId}')">수정</button>
                        <button class="action-btn delete" onclick="window.deleteReply('${qId}', '${replyId}')">삭제</button>
                    </div>
                </div>
                <div id="reply-text-${qId}-${replyId}" class="comment-text">${escapeHTML(data.text)}</div>
            </div>
        `;
        list.appendChild(li);
    }

    window.editReply = (qId, replyId) => {
        if (!isAdmin) return;
        const li = document.getElementById(`reply-${qId}-${replyId}`);
        const textContainer = document.getElementById(`reply-text-${qId}-${replyId}`);
        const currentText = li.dataset.text;

        textContainer.innerHTML = `
            <textarea id="edit-reply-input-${qId}-${replyId}" style="width:100%; height:60px; padding:8px; border:1px solid #1877f2; border-radius:6px; resize:none; font-family:inherit; outline:none; box-sizing:border-box; margin-top:5px;">${escapeHTML(currentText)}</textarea>
            <div style="display:flex; justify-content:flex-end; gap:5px; margin-top:5px;">
                <button onclick="window.cancelEditReply('${qId}', '${replyId}')" style="padding:4px 8px; font-size:12px; border:none; background:#e4e6eb; border-radius:4px; cursor:pointer;">취소</button>
                <button onclick="window.saveEditReply('${qId}', '${replyId}')" style="padding:4px 8px; font-size:12px; border:none; background:#1a73e8; color:white; border-radius:4px; cursor:pointer;">저장</button>
            </div>
        `;
    };

    window.cancelEditReply = (qId, replyId) => {
        const li = document.getElementById(`reply-${qId}-${replyId}`);
        const textContainer = document.getElementById(`reply-text-${qId}-${replyId}`);
        textContainer.textContent = li.dataset.text;
    };

    // ★ 답변 저장 오류 해결: 수동으로 화면 업데이트 ★
    window.saveEditReply = (qId, replyId) => {
        const newText = document.getElementById(`edit-reply-input-${qId}-${replyId}`).value.trim();
        if (newText) {
            update(ref(db, `questions/${qId}/replies/${replyId}`), { text: newText });
            
            // 데이터베이스 업데이트 후, 즉시 UI 원상복구 및 새 텍스트 반영
            const li = document.getElementById(`reply-${qId}-${replyId}`);
            li.dataset.text = newText;
            const textContainer = document.getElementById(`reply-text-${qId}-${replyId}`);
            textContainer.textContent = newText;
            
        } else {
            window.AppModal.show({type: 'alert', msg: '답변 내용을 입력해주세요.'});
        }
    };

    window.deleteReply = (qId, replyId) => {
        if (!isAdmin) return;
        window.AppModal.show({type: 'confirm', msg: '이 답변을 완전히 삭제하시겠습니까?', confirmText: '삭제'}, (res) => {
            if(res) remove(ref(db, `questions/${qId}/replies/${replyId}`));
        });
    };

    function getCurrentTime() {
        const now = new Date();
        const year = now.getFullYear();
        const month = String(now.getMonth() + 1).padStart(2, '0');
        const date = String(now.getDate()).padStart(2, '0');
        const ampm = now.getHours() >= 12 ? '오후' : '오전';
        let hour = now.getHours() % 12;
        hour = hour ? hour : 12;
        const minutes = String(now.getMinutes()).padStart(2, '0');
        return `${year}/${month}/${date} ${ampm} ${hour}:${minutes}`;
    }
